// get property value
// for each ... in
var sum = 0;
var obj = {prop1: 5, prop2: 13, prop3: 8};
for each (var item in obj) {
   sum += item;
}
document.write(sum); // prints "26", which is 5+13+8		    
// get property name
// for...in
var o = {a:1, b:2, c:3};
function show_props(obj, objName) {
  for (var i in obj) {
  document.write(objName + "." + i + " = " + obj[i] + "<br />");
}
}
show_props(o, "o");

// o.a = 1
// o.b = 2
// o.c = 3
    

iterator

get object property name and value as pair

var lang = { name: 'JavaScript', birthYear: 1995 };
var it = Iterator(lang); // pass object to Iterator

var pair = it.next(); // Pair is ["name", "JavaScript"]
document.write(pair);

pair = it.next(); // Pair is ["birthYear", 1995]
document.write(pair);

pair = it.next(); // A StopIteration exception is thrown
document.write(pair);

    

var lang = { name: 'JavaScript', birthYear: 1995 };
var it = Iterator(lang); // pass object to Iterator

for (var pair2 in it){
document.write(pair2+"<br />"); //[key, value] pair in turn
}

var it = Iterator(lang, true);
for (var key in it){
document.write(key+"<br />"); //each key in turn
}
