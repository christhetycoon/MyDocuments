// define constructor for object A
function A(a){
this.varA = a;
}

// define A's prototype
A.prototype = {
varA : null,
doSomething : function(){
alert("Hello");
}
}

// define constructor for object B, B will extend A
function B(a, b){
A.call(this, a); // call super constructor
this.varB = b;
}


// define B's prototype
B.prototype = {
varB : null,
doSomething : function(){ // override
A.prototype.doSomething.apply(this, arguments); // call super
}
}

// define extend function
function extend(child, supertype){
for (var property in supertype.prototype) {
if (typeof child.prototype[property] == "undefined")
child.prototype[property] = supertype.prototype[property];
}
return child;
}


extend(B, A);
var b = new B();
b.doSomething();

// In Ruby, all classes are mutable and can be changed at any point.
// This practice is referred to as “monkeypatching”

var Player = Class.create({
// initialize. Prototype treats it as the class’s constructor
initialize: function(firstName, lastName) {
this.firstName = firstName;
this.lastName = lastName;
this.points = 0;
},

functionName: function(argument){
// ...
},

functionName: function(argument){
// ...
}
});

var p = new Player('Andrew', 'Dupont');

Player.addMethods({
makeActive: function() {
this.isActive = true;
console.log(this + " will be a starter for Sunday's game.");
},

makeReserve: function() {
this.isActive = false;
console.log(this + " will spend Sunday on the bench.");
}
});

andrew.makeReserve(); // -> "Andrew Dupont will spend Sunday on the bench."
andrew.isActive; //-> false
